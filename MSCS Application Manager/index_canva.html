function refreshDeadlines() {
const deadlinesContainer = document.getElementById('deadlines-container');
const deadlines = JSON.parse(localStorage.getItem('deadlines'));
const universities = JSON.parse(localStorage.getItem('universities'));
deadlinesContainer.innerHTML = '';
if (deadlines.length === 0) {
deadlinesContainer.innerHTML = '<p class="text-gray-500 italic">No deadlines yet. Add one!</p>';
return;
}
// Sort deadlines by date
deadlines.sort((a, b) => new Date(a.date) - new Date(b.date));
// Only show upcoming deadlines (within 30 days)
const today = new Date();
const thirtyDaysLater = new Date();
thirtyDaysLater.setDate(today.getDate() + 30);
const upcomingDeadlines = deadlines.filter(deadline => {
const deadlineDate = new Date(deadline.date);
return deadlineDate >= today && deadlineDate <= thirtyDaysLater;
});
if (upcomingDeadlines.length === 0) {
deadlinesContainer.innerHTML = '<p class="text-gray-500 italic">No upcoming deadlines within 30 days.</p>';
return;
}
upcomingDeadlines.forEach(deadline => {
const deadlineElement = document.createElement('div');
deadlineElement.className = 'retro-container bg-[#f8f8f8] p-3 relative';
// Find associated university
let universityName = '';
if (deadline.universityId) {
const university = universities.find(uni => uni.id === deadline.universityId);
if (university) {
universityName = university.name;
}
}
// Calculate days remaining
const deadlineDate = new Date(deadline.date);
const daysRemaining = Math.ceil((deadlineDate - today) / (1000 * 60 * 60 * 24));
let urgencyClass = '';
if (daysRemaining <= 3) {
urgencyClass = 'text-red-600 font-bold';
} else if (daysRemaining <= 7) {
urgencyClass = 'text-orange-500 font-bold';
}
deadlineElement.innerHTML = `
<div class="flex justify-between items-start">
<div>
<h3 class="font-bold">${deadline.title}</h3>
${universityName ? `<p class="text-sm text-gray-600">${universityName}</p>` : ''}
</div>
<div class="text-right">
<div class="text-sm ${urgencyClass}">${formatDate(deadline.date)}</div>
<div class="text-xs ${urgencyClass}">${daysRemaining} day${daysRemaining !== 1 ? 's' : ''} left</div>
</div>
</div>
${deadline.notes ? `<p class="text-sm mt-2">${deadline.notes}</p>` : ''}
`;
deadlinesContainer.appendChild(deadlineElement);
});
}

function refreshTasks() {
const tasksContainer = document.getElementById('tasks-container');
const tasks = JSON.parse(localStorage.getItem('tasks'));
const universities = JSON.parse(localStorage.getItem('universities'));
tasksContainer.innerHTML = '';
if (tasks.length === 0) {
tasksContainer.innerHTML = '<p class="text-gray-500 italic">No tasks yet. Add one!</p>';
return;
}
// Sort tasks by due date and priority
tasks.sort((a, b) => {
if (a.completed !== b.completed) {
return a.completed ? 1 : -1;
}
if (a.dueDate !== b.dueDate) {
return new Date(a.dueDate) - new Date(b.dueDate);
}
const priorityOrder = { high: 0, medium: 1, low: 2 };
return priorityOrder[a.priority] - priorityOrder[b.priority];
});
tasks.forEach((task, index) => {
const taskElement = document.createElement('div');
taskElement.className = `retro-container bg-[#f8f8f8] p-3 relative ${task.completed ? 'opacity-60' : ''}`;
// Find associated university
let universityName = '';
if (task.universityId) {
const university = universities.find(uni => uni.id === task.universityId);
if (university) {
universityName = university.name;
}
}
// Priority indicator
let priorityColor = '';
if (task.priority === 'high') {
priorityColor = 'bg-red-400';
} else if (task.priority === 'medium') {
priorityColor = 'bg-yellow-400';
} else {
priorityColor = 'bg-green-400';
}
taskElement.innerHTML = `
<div class="flex items-start">
<input type="checkbox" class="task-checkbox mr-3 h-5 w-5 mt-1" data-id="${task.id}" ${task.completed ? 'checked' : ''}>
<div class="flex-grow">
<div class="flex justify-between items-start">
<div>
<p class="${task.completed ? 'line-through' : ''}">${task.description}</p>
${universityName ? `<p class="text-xs text-gray-600">${universityName}</p>` : ''}
</div>
<div class="flex items-center space-x-2">
<div class="w-3 h-3 rounded-full ${priorityColor}"></div>
<div class="text-xs">${formatDate(task.dueDate)}</div>
</div>
</div>
</div>
</div>
`;
tasksContainer.appendChild(taskElement);
// Add event listener to checkbox
taskElement.querySelector('.task-checkbox').addEventListener('change', (e) => {
toggleTaskCompletion(e.target.dataset.id);
});
});
}
function toggleTaskCompletion(taskId) {

tasks[taskIndex].updatedAt = new Date().toISOString();
localStorage.setItem('tasks', JSON.stringify(tasks));
// Add to history
addToHistory('tasks', tasks[taskIndex].completed ? 'completed' : 'uncompleted', tasks[taskIndex]);
refreshTasks();
}

const universitiesList = document.getElementById('universities-list');
const universities = JSON.parse(localStorage.getItem('universities'));
const filter = document.getElementById('university-filter').value;
const tagFilter = document.getElementById('university-tag-filter').value;
const searchTerm = document.getElementById('university-search').value.toLowerCase();
universitiesList.innerHTML = '';
if (universities.length === 0) {
universitiesList.innerHTML = '<p class="text-center py-4 text-gray-500 italic">No universities added yet.</p>';
return;
}
// Apply filters
const filteredUniversities = universities.filter(uni => {
const matchesStatus = filter === 'all' || uni.status === filter;
const matchesTag = tagFilter === 'all' || uni.tag === tagFilter;
const matchesSearch = uni.name.toLowerCase().includes(searchTerm) ||
uni.program.toLowerCase().includes(searchTerm);
return matchesStatus && matchesTag && matchesSearch;
});
if (filteredUniversities.length === 0) {
universitiesList.innerHTML = '<p class="text-center py-4 text-gray-500 italic">No universities match your filters.</p>';
return;
}
// Sort universities by name
filteredUniversities.sort((a, b) => a.name.localeCompare(b.name));
filteredUniversities.forEach(uni => {
const uniElement = document.createElement('div');
uniElement.className = 'grid grid-cols-12 gap-4 py-3 border-b border-gray-200 items-center';
// Status badge
let statusClass = '';
switch (uni.status) {
case 'researching': statusClass = 'bg-gray-200'; break;
case 'applying': statusClass = 'bg-blue-200'; break;
case 'applied': statusClass = 'bg-yellow-200'; break;
case 'admitted': statusClass = 'bg-green-200'; break;
case 'rejected': statusClass = 'bg-red-200'; break;
case 'waitlisted': statusClass = 'bg-purple-200'; break;
}
// Tag badge
let tagClass = '';
switch (uni.tag) {
case 'reach': tagClass = 'tag-reach'; break;
case 'target': tagClass = 'tag-target'; break;
case 'safety': tagClass = 'tag-safety'; break;
}
uniElement.innerHTML = `
<div class="col-span-3">
<p class="font-semibold">${uni.name}</p>
</div>
<div class="col-span-2">
<p>${uni.program}</p>
</div>
<div class="col-span-2">
<span class="px-2 py-1 rounded text-sm ${statusClass}">${capitalizeFirstLetter(uni.status)}</span>
</div>
<div class="col-span-1">
<span class="px-2 py-1 rounded text-sm ${tagClass}">${capitalizeFirstLetter(uni.tag)}</span>
</div>
<div class="col-span-2">
<p class="text-sm">${formatDate(uni.deadline)}</p>
</div>
<div class="col-span-2 flex space-x-2">
<button class="view-university retro-btn bg-white px-2 py-1 text-xs" data-id="${uni.id}">View</button>
<button class="edit-university retro-btn bg-[#c8b6ff] px-2 py-1 text-xs" data-id="${uni.id}">Edit</button>
</div>
`;
universitiesList.appendChild(uniElement);
});
// Add event listeners to view and edit buttons
document.querySelectorAll('.view-university').forEach(btn => {
btn.addEventListener('click', (e) => {
viewUniversity(e.target.dataset.id);
});
});
document.querySelectorAll('.edit-university').forEach(btn => {
btn.addEventListener('click', (e) => {
editUniversity(e.target.dataset.id);
});
});
}
function viewUniversity(universityId) {
const universities = JSON.parse(localStorage.getItem('universities'));
const requirements = JSON.parse(localStorage.getItem('requirements'));
const documents = JSON.parse(localStorage.getItem('documents'));
const university = universities.find(uni => uni.id === universityId);
if (!university) return;
// Populate university details
document.getElementById('uni-detail-name').textContent = university.name;
document.getElementById('uni-detail-program').textContent = university.program;
document.getElementById('uni-detail-status').textContent = capitalizeFirstLetter(university.status);
// Set tag with appropriate class
const tagElement = document.getElementById('uni-detail-tag');
tagElement.textContent = capitalizeFirstLetter(university.tag);
tagElement.className = 'p-2';
switch (university.tag) {
case 'reach': tagElement.classList.add('tag-reach'); break;
case 'target': tagElement.classList.add('tag-target'); break;
case 'safety': tagElement.classList.add('tag-safety'); break;
}
document.getElementById('uni-detail-deadline').textContent = formatDate(university.deadline);
document.getElementById('uni-detail-url').textContent = university.url || 'No URL provided';
document.getElementById('uni-detail-notes').textContent = university.notes || 'No notes added.';
// Find requirements for this university
const uniRequirements = requirements.find(req => req.universityId === universityId);
if (uniRequirements) {
document.getElementById('uni-detail-requirements').innerHTML = `
<div class="space-y-2">
${uniRequirements.gre ? `<p><strong>GRE:</strong> ${uniRequirements.gre}</p>` : ''}
${uniRequirements.toefl ? `<p><strong>TOEFL/IELTS:</strong> ${uniRequirements.toefl}</p>` : ''}
${uniRequirements.gpa ? `<p><strong>GPA:</strong> ${uniRequirements.gpa}</p>` : ''}
${uniRequirements.background ? `<p><strong>Background:</strong> ${uniRequirements.background}</p>` : ''}
${uniRequirements.documents ? `<p><strong>Required Documents:</strong> ${uniRequirements.documents}</p>` : ''}
${uniRequirements.opt !== 'unknown' ? `<p><strong>OPT/H1B Eligibility:</strong> ${uniRequirements.opt === 'yes' ? 'Yes' : 'No'}</p>` : ''}
${uniRequirements.stem !== 'unknown' ? `<p><strong>STEM Designation:</strong> ${uniRequirements.stem === 'yes' ? 'Yes' : 'No'}</p>` : ''}
${uniRequirements.jobInfo ? `<p><strong>Job Placement Info:</strong> ${uniRequirements.jobInfo}</p>` : ''}
</div>
`;
} else {
document.getElementById('uni-detail-requirements').innerHTML = '<p class="text-gray-500 italic">No requirements added yet.</p>';
}
// Find documents associated with this university
const associatedDocs = documents.filter(doc => doc.universities.includes(universityId));
if (associatedDocs.length > 0) {
document.getElementById('uni-detail-documents').innerHTML = associatedDocs.map(doc => {
return `<div class="mb-1"><span class="font-semibold">${getDocumentTypeLabel(doc.type)}:</span> ${doc.name}</div>`;
}).join('');
} else {
document.getElementById('uni-detail-documents').innerHTML = '<p class="text-gray-500 italic">No documents associated yet.</p>';
}
// Set up delete and edit buttons
document.getElementById('delete-university-btn').dataset.id = universityId;
document.getElementById('edit-university-btn').dataset.id = universityId;
// Show modal
document.getElementById('university-details-modal').classList.remove('hidden');
// Add event listeners
document.getElementById('delete-university-btn').addEventListener('click', () => {
if (confirm('Are you sure you want to delete this university? This action cannot be undone.')) {
deleteUniversity(universityId);
document.getElementById('university-details-modal').classList.add('hidden');
}
});
document.getElementById('edit-university-btn').addEventListener('click', () => {
document.getElementById('university-details-modal').classList.add('hidden');
editUniversity(universityId);
});
}
